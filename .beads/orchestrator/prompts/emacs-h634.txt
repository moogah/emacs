(defun jf/bash--extract-for-loop-structure (for-node)
  "Extract loop structure from FOR-NODE.
FOR-NODE is a for_statement tree-sitter node.
Returns plist with loop metadata."
  (let* ((variable-node (treesit-node-child-by-field-name for-node "variable"))
         (variable-name (when variable-node
                         (treesit-node-text variable-node t)))
         (body-text nil)
         (iteration-values nil)
         (command-substitutions nil))
    ;; ... shared extraction logic ...
    (list :success t
          :type :for-loop
          :command-name "for"
          :variable variable-name
          :iteration-values iteration-values
          :body-text body-text
          :loop-variable variable-name
          :loop-body body-text
          :loop-list (car iteration-values)
          :loop-source-type loop-source-type
          :ast for-node)))

(defun jf/bash-parse--handle-for-loop (root-node)
  "Handle for-loop from ROOT-NODE by finding node and delegating."
  (let ((for-node (jf/bash-parse--find-node-by-type root-node "for_statement")))
    (if for-node
        (jf/bash--extract-for-loop-structure for-node)
      (list :success nil :error "No for_statement found"))))

(defun jf/bash-parse--handle-for-loop-node (for-node)
  "Handle for-loop from FOR-NODE directly."
  (jf/bash--extract-for-loop-structure for-node))

Extract shared helper:
```elisp
(defun jf/bash--resolve-directory-target (target var-context current-pwd &optional allow-no-args)
  "Resolve directory TARGET for cd/pushd commands.
TARGET is the target string.
VAR-CONTEXT is variable bindings.
CURRENT-PWD is current directory.
ALLOW-NO-ARGS controls behavior when target is nil.

Returns resolved absolute path or :unresolved marker."
  (cond
   ;; No target - handle based on allow-no-args
   ((null target)
    (if allow-no-args
        (or (alist-get 'HOME var-context) :unresolved)
      :unresolved))
   
   ;; Empty string
   ((string-empty-p target) :unresolved)
   
   ;; cd - (OLDPWD)
   ((equal target "-")
    (or (alist-get 'OLDPWD var-context) :unresolved))
   
   ;; cd ~ (HOME)
   ((equal target "~")
    (or (alist-get 'HOME var-context) :unresolved))
   
   ;; cd ~/path (HOME expansion)
   ((string-prefix-p "~/" target)
    (if-let ((home (alist-get 'HOME var-context)))
        (expand-file-name (concat home "/" (substring target 2)))
      :unresolved))
   
   ;; Regular target - resolve variables, command substitutions, relative paths
   (t
    (let ((var-resolved (jf/bash-resolve-variables target var-context)))
      (if (and (listp var-resolved) (plist-get var-resolved :unresolved))
          :unresolved
        (let ((cmd-resolved (jf/bash-resolve-command-substitution 
                            var-resolved var-context)))
          (if (eq cmd-resolved :unresolved)
              :unresolved
            (jf/bash-resolve-relative-path cmd-resolved var-context current-pwd))))))))

(defun jf/bash--extract-cd-target (command var-context current-pwd)
  "Extract and resolve cd target using shared logic."
  (let ((target (jf/bash--get-cd-argument command)))
    (jf/bash--resolve-directory-target target var-context current-pwd t)))

(defun jf/bash--extract-pushd-target (command var-context current-pwd)
  "Extract and resolve pushd target using shared logic."
  (let ((target (jf/bash--get-pushd-argument command)))
    (jf/bash--resolve-directory-target target var-context current-pwd nil)))
```
